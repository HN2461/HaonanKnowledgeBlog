# MongoDB 详解第七篇：完整项目实战

> 把前面六篇的知识全部串起来，从零搭一个真实可用的笔记管理 API。包含：用户注册/登录、JWT 鉴权、笔记增删改查、分页搜索、软删除。

## 一、项目规划

### 1.1 功能清单

```
用户模块：
  POST /api/auth/register  — 注册
  POST /api/auth/login     — 登录

笔记模块（需要登录）：
  GET    /api/notes        — 获取笔记列表（分页 + 搜索）
  GET    /api/notes/:id    — 获取单篇笔记
  POST   /api/notes        — 创建笔记
  PUT    /api/notes/:id    — 更新笔记
  DELETE /api/notes/:id    — 删除笔记（软删除）
```

### 1.2 项目结构

```
notes-api/
├── src/
│   ├── models/
│   │   ├── User.js       ← 用户 Model
│   │   └── Note.js       ← 笔记 Model
│   ├── routes/
│   │   ├── auth.js       ← 注册/登录路由
```

## 三、定义 Model

### 3.1 User Model

```javascript
// src/models/User.js
const { Schema, model } = require('mongoose')
const bcrypt = require('bcrypt')

const userSchema = new Schema({
  username: {
    type: String,
    required: [true, '用户名不能为空'],
    trim: true,
    minlength: [2, '用户名至少2个字符'],
    maxlength: [20, '用户名最多20个字符']
  },
  email: {
    type: String,
    required: [true, '邮箱不能为空'],
    unique: true,
    lowercase: true,
    trim: true,
    match: [/^\S+@\S+\.\S+$/, '邮箱格式不正确']
  },
  password: {
    type: String,
    required: [true, '密码不能为空'],
    minlength: [6, '密码至少6位']
  }
}, {
  timestamps: true,
  versionKey: false
})

// 保存前自动加密密码
userSchema.pre('save', async function() {
  // 只有 password 被修改时才重新加密
  if (!this.isModified('password')) return
  this.password = await bcrypt.hash(this.password, 10)
})

// 实例方法：验证密码
userSchema.methods.comparePassword = async function(plainPassword) {
  return bcrypt.compare(plainPassword, this.password)
}

// toJSON 时移除 password
userSchema.set('toJSON', {
  transform: function(doc, ret) {
    delete ret.password
    return ret
  }
})

module.exports = model('User', userSchema)
```

### 3.2 Note Model

```javascript
// src/models/Note.js
const { Schema, model } = require('mongoose')

const noteSchema = new Schema({
  title: {
    type: String,
    required: [true, '标题不能为空'],
    trim: true,
    maxlength: [100, '标题最多100个字符']
  },
  content: {
    type: String,
    default: ''
  },
  tags: {
    type: [String],
    default: []
  },
  // 关联用户
  author: {
    type: Schema.Types.ObjectId,
    ref: 'User',
    required: true
  },
  // 软删除字段
  isDeleted: {
    type: Boolean,
    default: false
  },
  deletedAt: Date
}, {
  timestamps: true,
  versionKey: false
})

// 复合索引：按作者 + 创建时间查询（最常见的查询模式）
noteSchema.index({ author: 1, createdAt: -1 })

// 文本索引：支持标题和内容的全文搜索
noteSchema.index({ title: 'text', content: 'text' })

// 查询前自动过滤软删除
noteSchema.pre(['find', 'findOne', 'findOneAndUpdate', 'countDocuments'], function() {
  const filter = this.getFilter()
  if (filter.isDeleted === undefined) {
    this.where({ isDeleted: { $ne: true } })
  }
})

module.exports = model('Note', noteSchema)
```

## 四、JWT 鉴权中间件

```javascript
// src/middleware/auth.js
const jwt = require('jsonwebtoken')

// JWT 密钥（生产环境放在环境变量里）
const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key-change-in-production'

// 生成 token 的工具函数（在路由里用）
function generateToken(userId) {
  return jwt.sign(
    { userId },           // payload：存放用户 ID
    JWT_SECRET,
    { expiresIn: '7d' }   // 7天过期
  )
}

// 鉴权中间件：验证请求头里的 token
function authMiddleware(req, res, next) {
  // 从请求头获取 token
  // 格式：Authorization: Bearer <token>
  const authHeader = req.headers.authorization
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ code: 401, message: '未登录，请先登录' })
  }

  const token = authHeader.split(' ')[1]

  try {
    // 验证 token，解析出 payload
    const decoded = jwt.verify(token, JWT_SECRET)
    // 把用户 ID 挂到 req 上，后续路由可以用
    req.userId = decoded.userId
    next()
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).json({ code: 401, message: 'token 已过期，请重新登录' })
    }
    return res.status(401).json({ code: 401, message: 'token 无效' })
  }
}

module.exports = { authMiddleware, generateToken }
```

## 五、注册/登录路由

```javascript
// src/routes/auth.js
const express = require('express')
const router = express.Router()
const User = require('../models/User')
const { generateToken } = require('../middleware/auth')

// POST /api/auth/register — 注册
router.post('/register', async (req, res) => {
  try {
    const { username, email, password } = req.body

    // 基础参数校验
    if (!username || !email || !password) {
      return res.status(400).json({ code: 400, message: '用户名、邮箱和密码不能为空' })
    }

    // 创建用户（密码在 pre save 钩子里自动加密）
    const user = await User.create({ username, email, password })

    // 生成 token
    const token = generateToken(user._id)

    res.status(201).json({
      code: 201,
      message: '注册成功',
      data: {
        token,
        user: user.toJSON()  // toJSON 会自动移除 password
      }
    })

  } catch (err) {
    // 邮箱重复
    if (err.code === 11000) {
      return res.status(409).json({ code: 409, message: '该邮箱已被注册' })
    }
    // Schema 校验失败
    if (err.name === 'ValidationError') {
      const messages = Object.values(err.errors).map(e => e.message)
      return res.status(400).json({ code: 400, message: messages.join('；') })
    }
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

// POST /api/auth/login — 登录
router.post('/login', async (req, res) => {
  try {
    const { email, password } = req.body

    if (!email || !password) {
      return res.status(400).json({ code: 400, message: '邮箱和密码不能为空' })
    }

    // 查找用户（注意：toJSON 会移除 password，这里需要 password 来验证）
    // 所以用 select('+password') 或者不用 lean()
    const user = await User.findOne({ email })
    if (!user) {
      // 不要说"用户不存在"，统一说"邮箱或密码错误"（安全考虑）
      return res.status(401).json({ code: 401, message: '邮箱或密码错误' })
    }

    // 验证密码
    const isMatch = await user.comparePassword(password)
    if (!isMatch) {
      return res.status(401).json({ code: 401, message: '邮箱或密码错误' })
    }

    // 生成 token
    const token = generateToken(user._id)

    res.json({
      code: 200,
      message: '登录成功',
      data: {
        token,
        user: user.toJSON()
      }
    })

  } catch (err) {
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

module.exports = router
```

## 六、笔记 CRUD 路由

```javascript
// src/routes/notes.js
const express = require('express')
const router = express.Router()
const Note = require('../models/Note')
const { authMiddleware } = require('../middleware/auth')

// 所有笔记路由都需要登录
router.use(authMiddleware)

// ─────────────────────────────────────────
// GET /api/notes — 获取笔记列表（分页 + 搜索）
// ─────────────────────────────────────────
router.get('/', async (req, res) => {
  try {
    const {
      page = 1,
      limit = 10,
      keyword,   // 搜索关键词
      tag        // 按标签筛选
    } = req.query

    const pageNum = Math.max(1, parseInt(page))
    const limitNum = Math.min(50, Math.max(1, parseInt(limit)))  // 最多50条

    // 构建查询条件（只查当前用户的笔记）
    const filter = { author: req.userId }

    // 关键词搜索（标题 + 内容）
    if (keyword && keyword.trim()) {
      filter.$or = [
        { title: { $regex: keyword.trim(), $options: 'i' } },
        { content: { $regex: keyword.trim(), $options: 'i' } }
      ]
    }

    // 标签筛选
    if (tag) {
      filter.tags = tag
    }

    // 并行查询数据和总数
    const [notes, total] = await Promise.all([
      Note.find(filter)
        .select('title tags createdAt updatedAt')  // 列表页不返回 content（节省流量）
        .sort({ createdAt: -1 })
        .skip((pageNum - 1) * limitNum)
        .limit(limitNum)
        .lean(),
      Note.countDocuments(filter)
    ])

    res.json({
      code: 200,
      data: {
        list: notes,
        pagination: {
          page: pageNum,
          limit: limitNum,
          total,
          pages: Math.ceil(total / limitNum)
        }
      }
    })

  } catch (err) {
    res.status(500).json({ code: 500, message: err.message })
  }
})

// ─────────────────────────────────────────
// GET /api/notes/:id — 获取单篇笔记
// ─────────────────────────────────────────
router.get('/:id', async (req, res) => {
  try {
    const note = await Note.findOne({
      _id: req.params.id,
      author: req.userId  // 只能查自己的笔记
    }).lean()

    if (!note) {
      return res.status(404).json({ code: 404, message: '笔记不存在' })
    }

    res.json({ code: 200, data: note })

  } catch (err) {
    if (err.name === 'CastError') {
      return res.status(400).json({ code: 400, message: 'ID 格式不正确' })
    }
    res.status(500).json({ code: 500, message: err.message })
  }
})

// ─────────────────────────────────────────
// POST /api/notes — 创建笔记
// ─────────────────────────────────────────
router.post('/', async (req, res) => {
  try {
    const { title, content, tags } = req.body

    const note = await Note.create({
      title,
      content,
      tags: Array.isArray(tags) ? tags : [],
      author: req.userId  // 从 token 里取用户 ID
    })

    res.status(201).json({
      code: 201,
      message: '创建成功',
      data: note
    })

  } catch (err) {
    if (err.name === 'ValidationError') {
      const messages = Object.values(err.errors).map(e => e.message)
      return res.status(400).json({ code: 400, message: messages.join('；') })
    }
    res.status(500).json({ code: 500, message: err.message })
  }
})

// ─────────────────────────────────────────
// PUT /api/notes/:id — 更新笔记
// ─────────────────────────────────────────
router.put('/:id', async (req, res) => {
  try {
    const { title, content, tags } = req.body

    // 构建更新对象（只更新传入的字段）
    const updateData = {}
    if (title !== undefined) updateData.title = title
    if (content !== undefined) updateData.content = content
    if (tags !== undefined) updateData.tags = Array.isArray(tags) ? tags : []

    const note = await Note.findOneAndUpdate(
      {
        _id: req.params.id,
        author: req.userId  // 只能更新自己的笔记
      },
      { $set: updateData },
      {
        new: true,           // 返回更新后的文档
        runValidators: true  // 执行 Schema 校验
      }
    )

    if (!note) {
      return res.status(404).json({ code: 404, message: '笔记不存在' })
    }

    res.json({ code: 200, message: '更新成功', data: note })

  } catch (err) {
    if (err.name === 'CastError') {
      return res.status(400).json({ code: 400, message: 'ID 格式不正确' })
    }
    if (err.name === 'ValidationError') {
      const messages = Object.values(err.errors).map(e => e.message)
      return res.status(400).json({ code: 400, message: messages.join('；') })
    }
    res.status(500).json({ code: 500, message: err.message })
  }
})

// ─────────────────────────────────────────
// DELETE /api/notes/:id — 删除笔记（软删除）
// ─────────────────────────────────────────
router.delete('/:id', async (req, res) => {
  try {
    // 软删除：标记 isDeleted = true，不真正删除数据
    const note = await Note.findOneAndUpdate(
      {
        _id: req.params.id,
        author: req.userId
      },
      {
        $set: {
          isDeleted: true,
          deletedAt: new Date()
        }
      },
      { new: true }
    )

    if (!note) {
      return res.status(404).json({ code: 404, message: '笔记不存在' })
    }

    // 204 No Content — 删除成功，不返回内容
    res.status(204).send()

  } catch (err) {
    if (err.name === 'CastError') {
      return res.status(400).json({ code: 400, message: 'ID 格式不正确' })
    }
    res.status(500).json({ code: 500, message: err.message })
  }
})

module.exports = router
```

## 七、入口文件

```javascript
// index.js
const express = require('express')
const connectDB = require('./src/db')
const authRoutes = require('./src/routes/auth')
const notesRoutes = require('./src/routes/notes')

const app = express()

// 解析 JSON 请求体
app.use(express.json())

// 路由
app.use('/api/auth', authRoutes)

// 404 处理
app.use((req, res) => {
  res.status(404).json({ code: 404, message: '接口不存在' })
})

// 全局错误处理（兜底）
app.use((err, req, res, next) => {
  console.error(err.stack)
  res.status(500).json({ code: 500, message: '服务器内部错误' })
})

// 先连接数据库，再启动服务器
connectDB().then(() => {
  app.listen(3000, () => {
    console.log('🚀 服务器运行在 http://localhost:3000')
  })
})
```

## 十、小结与进阶方向

这个项目覆盖了 MongoDB + Mongoose 的核心用法。如果想继续深入，可以探索：

```
进阶方向：
1. 环境变量管理 — 用 dotenv 管理 JWT_SECRET、数据库连接字符串
2. 参数校验 — 用 joi 或 express-validator 做更严格的入参校验
3. 限流 — 用 express-rate-limit 防止暴力破解
4. 日志 — 用 morgan 记录请求日志
5. 文件上传 — 用 multer + GridFS 存储附件
6. 全文搜索 — MongoDB Atlas Search 支持中文分词
7. 缓存 — Redis 缓存热点数据，减少数据库压力
8. 部署 — MongoDB Atlas（云数据库）+ Railway/Render 部署
```
