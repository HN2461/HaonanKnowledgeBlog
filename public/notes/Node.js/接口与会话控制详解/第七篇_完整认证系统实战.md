# 接口与会话控制详解第七篇：完整认证系统实战

> 把前面六篇的知识全部串起来，从零搭一个完整的认证系统。这个项目可以直接作为你下一个项目的认证模板。

---

## 一、项目规划

### 1.1 功能清单

```
认证模块：
  POST /api/auth/register   — 注册（参数校验 + 限流）
  POST /api/auth/login      — 登录（限流 + 双 Token）
  POST /api/auth/refresh    — 刷新 access token
  POST /api/auth/logout     — 退出登录（吊销 refresh token）

用户模块（需要登录）：
  GET  /api/users/me        — 获取当前用户信息
  PUT  /api/users/me        — 更新当前用户信息

管理员模块（需要 admin 角色）：
  GET  /api/admin/users     — 获取用户列表
  PUT  /api/admin/users/:id/role  — 修改用户角色
```

## 四、Model 定义

### 4.1 User Model

```javascript
// src/models/User.js
const { Schema, model } = require('mongoose')
const bcrypt = require('bcrypt')

const userSchema = new Schema({
  username: {
    type: String,
    required: [true, '用户名不能为空'],
    unique: true,
    trim: true,
    minlength: [2, '用户名至少2个字符'],
    maxlength: [20, '用户名最多20个字符']
  },
  email: {
    type: String,
    required: [true, '邮箱不能为空'],
    unique: true,
    lowercase: true,
    trim: true,
    match: [/^\S+@\S+\.\S+$/, '邮箱格式不正确']
  },
  password: {
    type: String,
    required: [true, '密码不能为空'],
    minlength: [6, '密码至少6位']
  },
  role: {
    type: String,
    enum: ['user', 'admin'],
    default: 'user'
  },
  isActive: {
    type: Boolean,
    default: true
  }
}, {
  timestamps: true,
  versionKey: false
})

// 保存前加密密码
userSchema.pre('save', async function() {
  if (!this.isModified('password')) return
  this.password = await bcrypt.hash(this.password, 10)
})

// 验证密码
userSchema.methods.comparePassword = async function(plain) {
  return bcrypt.compare(plain, this.password)
}

// toJSON 时移除敏感字段
userSchema.set('toJSON', {
  transform(doc, ret) {
    delete ret.password
    return ret
  }
})

module.exports = model('User', userSchema)
```

## 五、中间件

### 5.1 JWT 鉴权中间件

```javascript
// src/middleware/auth.js
const jwt = require('jsonwebtoken')

// 鉴权中间件
function authMiddleware(req, res, next) {
  const authHeader = req.headers.authorization

  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ code: 401, message: '未登录，请先登录' })
  }

  const token = authHeader.split(' ')[1]

  try {
    const payload = jwt.verify(token, process.env.ACCESS_TOKEN_SECRET)
    req.user = payload
    next()
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).json({ code: 401, message: 'token 已过期，请刷新' })
    }
    return res.status(401).json({ code: 401, message: 'token 无效' })
  }
}

// 角色权限中间件
function requireRole(...roles) {
  return (req, res, next) => {
    if (!roles.includes(req.user?.role)) {
      return res.status(403).json({ code: 403, message: '权限不足' })
    }
    next()
  }
}

module.exports = { authMiddleware, requireRole }
```

### 5.2 校验结果处理中间件

```javascript
// src/middleware/validate.js
const { validationResult } = require('express-validator')

function validate(req, res, next) {
  const errors = validationResult(req)
  if (!errors.isEmpty()) {
    return res.status(400).json({
      code: 400,
      message: errors.array()[0].msg,
      data: null
    })
  }
  next()
}

module.exports = validate
```

## 七、路由实现

### 7.1 认证路由

```javascript
// src/routes/auth.js
const express = require('express')
const router = express.Router()
const jwt = require('jsonwebtoken')
const rateLimit = require('express-rate-limit')
const User = require('../models/User')
const RefreshToken = require('../models/RefreshToken')
const validate = require('../middleware/validate')
const { registerRules, loginRules } = require('../validators/auth')
const { authMiddleware } = require('../middleware/auth')

// 限流配置
const loginLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 5,
  message: { code: 429, message: '登录尝试次数过多，请 15 分钟后再试' }
})

const registerLimiter = rateLimit({
  windowMs: 60 * 60 * 1000,
  max: 5,
  message: { code: 429, message: '注册过于频繁，请稍后再试' }
})

// 生成双 Token
function generateTokens(user) {
  const payload = { userId: user._id, username: user.username, role: user.role }

  const accessToken = jwt.sign(payload, process.env.ACCESS_TOKEN_SECRET, {
    expiresIn: process.env.ACCESS_TOKEN_EXPIRES || '15m'
  })

  const refreshToken = jwt.sign(
    { userId: user._id },
    process.env.REFRESH_TOKEN_SECRET,
    { expiresIn: process.env.REFRESH_TOKEN_EXPIRES || '7d' }
  )

  return { accessToken, refreshToken }
}

// ─── 注册 ───
router.post('/register', registerLimiter, registerRules, validate, async (req, res) => {
  try {
    const { username, email, password } = req.body

    const user = await User.create({ username, email, password })

    const { accessToken, refreshToken } = generateTokens(user)

    // 保存 refresh token
    await RefreshToken.create({
      token: refreshToken,
      userId: user._id,
      expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000)
    })

    res.status(201).json({
      code: 201,
      message: '注册成功',
      data: { accessToken, refreshToken, user: user.toJSON() }
    })

  } catch (err) {
    if (err.code === 11000) {
      const field = Object.keys(err.keyValue)[0]
      return res.status(409).json({ code: 409, message: `${field === 'email' ? '邮箱' : '用户名'}已被注册` })
    }
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

// ─── 登录 ───
router.post('/login', loginLimiter, loginRules, validate, async (req, res) => {
  try {
    const { email, password } = req.body

    const user = await User.findOne({ email })
    if (!user || !(await user.comparePassword(password))) {
      return res.status(401).json({ code: 401, message: '邮箱或密码错误' })
    }

    if (!user.isActive) {
      return res.status(403).json({ code: 403, message: '账号已被禁用' })
    }

    const { accessToken, refreshToken } = generateTokens(user)

    await RefreshToken.create({
      token: refreshToken,
      userId: user._id,
      expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000)
    })

    res.json({
      code: 200,
      message: '登录成功',
      data: { accessToken, refreshToken, user: user.toJSON() }
    })

  } catch (err) {
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

// ─── 刷新 access token ───
router.post('/refresh', async (req, res) => {
  const { refreshToken } = req.body

  if (!refreshToken) {
    return res.status(401).json({ code: 401, message: '缺少 refresh token' })
  }

  try {
    const payload = jwt.verify(refreshToken, process.env.REFRESH_TOKEN_SECRET)

    const stored = await RefreshToken.findOne({ token: refreshToken, userId: payload.userId })
    if (!stored) {
      return res.status(401).json({ code: 401, message: 'refresh token 无效或已被吊销' })
    }

    const user = await User.findById(payload.userId)
    if (!user || !user.isActive) {
      return res.status(401).json({ code: 401, message: '用户不存在或已被禁用' })
    }

    const accessToken = jwt.sign(
      { userId: user._id, username: user.username, role: user.role },
      process.env.ACCESS_TOKEN_SECRET,
      { expiresIn: process.env.ACCESS_TOKEN_EXPIRES || '15m' }
    )

    res.json({ code: 200, data: { accessToken } })

  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).json({ code: 401, message: 'refresh token 已过期，请重新登录' })
    }
    res.status(401).json({ code: 401, message: 'refresh token 无效' })
  }
})

// ─── 退出登录 ───
router.post('/logout', authMiddleware, async (req, res) => {
  const { refreshToken } = req.body

  if (refreshToken) {
    await RefreshToken.deleteOne({ token: refreshToken, userId: req.user.userId })
  }

  res.json({ code: 200, message: '已退出登录' })
})

module.exports = router
```

### 7.2 用户路由

```javascript
// src/routes/users.js
const express = require('express')
const router = express.Router()
const { body } = require('express-validator')
const User = require('../models/User')
const { authMiddleware } = require('../middleware/auth')
const validate = require('../middleware/validate')

// 所有路由都需要登录
router.use(authMiddleware)

// GET /api/users/me — 获取当前用户信息
router.get('/me', async (req, res) => {
  try {
    const user = await User.findById(req.user.userId).lean()
    if (!user) return res.status(404).json({ code: 404, message: '用户不存在' })

    // lean() 返回的对象没有 toJSON，手动删除 password
    delete user.password
    res.json({ code: 200, data: user })
  } catch (err) {
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

// PUT /api/users/me — 更新当前用户信息
const updateRules = [
  body('username').optional().trim()
    .isLength({ min: 2, max: 20 }).withMessage('用户名长度 2-20 个字符'),
  body('bio').optional()
    .isLength({ max: 200 }).withMessage('简介最多200字'),
  // 不允许通过这个接口修改 role 和 email
]

router.put('/me', updateRules, validate, async (req, res) => {
  try {
    // 只允许更新白名单字段
    const { username, bio } = req.body
    const updateData = {}
    if (username !== undefined) updateData.username = username
    if (bio !== undefined) updateData.bio = bio

    const user = await User.findByIdAndUpdate(
      req.user.userId,
      { $set: updateData },
      { new: true, runValidators: true }
    )

    if (!user) return res.status(404).json({ code: 404, message: '用户不存在' })

    res.json({ code: 200, message: '更新成功', data: user.toJSON() })
  } catch (err) {
    if (err.code === 11000) {
      return res.status(409).json({ code: 409, message: '用户名已存在' })
    }
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

module.exports = router
```

### 7.3 管理员路由

```javascript
// src/routes/admin.js
const express = require('express')
const router = express.Router()
const User = require('../models/User')
const { authMiddleware, requireRole } = require('../middleware/auth')

// 所有管理员路由：需要登录 + admin 角色
router.use(authMiddleware, requireRole('admin'))

// GET /api/admin/users — 获取用户列表
router.get('/users', async (req, res) => {
  try {
    const { page = 1, limit = 20, keyword } = req.query

    const filter = {}
    if (keyword) {
      filter.$or = [
        { username: { $regex: keyword, $options: 'i' } },
        { email: { $regex: keyword, $options: 'i' } }
      ]
    }

    const [users, total] = await Promise.all([
      User.find(filter)
        .select('-password')
        .sort({ createdAt: -1 })
        .skip((page - 1) * limit)
        .limit(Number(limit))
        .lean(),
      User.countDocuments(filter)
    ])

    res.json({
      code: 200,
      data: {
        list: users,
        pagination: { page: Number(page), limit: Number(limit), total }
      }
    })
  } catch (err) {
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

// PUT /api/admin/users/:id/role — 修改用户角色
router.put('/users/:id/role', async (req, res) => {
  try {
    const { role } = req.body

    if (!['user', 'admin'].includes(role)) {
      return res.status(400).json({ code: 400, message: '角色值不合法' })
    }

    // 不能修改自己的角色
    if (req.params.id === req.user.userId) {
      return res.status(400).json({ code: 400, message: '不能修改自己的角色' })
    }

    const user = await User.findByIdAndUpdate(
      req.params.id,
      { $set: { role } },
      { new: true }
    )

    if (!user) return res.status(404).json({ code: 404, message: '用户不存在' })

    res.json({ code: 200, message: '角色已更新', data: user.toJSON() })
  } catch (err) {
    res.status(500).json({ code: 500, message: '服务器错误' })
  }
})

module.exports = router
```

## 八、入口文件

```javascript
// index.js
require('dotenv').config()

const express = require('express')
const helmet = require('helmet')
const rateLimit = require('express-rate-limit')
const connectDB = require('./src/db')

const authRoutes = require('./src/routes/auth')
const userRoutes = require('./src/routes/users')
const adminRoutes = require('./src/routes/admin')

const app = express()

// 安全响应头
app.use(helmet())

// 关闭 X-Powered-By
app.disable('x-powered-by')

// 解析 JSON
app.use(express.json())

// 全局限流
app.use(rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 200,
  message: { code: 429, message: '请求过于频繁，请稍后再试' }
}))

// 路由
app.use('/api/auth', authRoutes)
app.use('/api/users', userRoutes)
app.use('/api/admin', adminRoutes)

// 404
app.use((req, res) => {
  res.status(404).json({ code: 404, message: '接口不存在' })
})

// 全局错误处理
app.use((err, req, res, next) => {
  console.error(err.stack)
  res.status(500).json({ code: 500, message: '服务器内部错误' })
})

const PORT = process.env.PORT || 3000

connectDB().then(() => {
  app.listen(PORT, () => {
    console.log(`🚀 服务器运行在 http://localhost:${PORT}`)
  })
})
```

```bash
node index.js
```

```bash
curl -X POST http://localhost:3000/api/auth/register \
```

```bash
curl -X POST http://localhost:3000/api/auth/login \
```

```bash
curl http://localhost:3000/api/users/me \
```

```bash
curl -X POST http://localhost:3000/api/auth/refresh \
```

```bash
curl -X POST http://localhost:3000/api/auth/logout \
  -H "Authorization: Bearer <accessToken>" \
  -H "Content-Type: application/json" \
  -d '{"refreshToken":"<refreshToken>"}'
```

## 十一、小结与进阶方向

这套认证系统覆盖了生产环境的基本需求。如果想继续完善：

```
进阶方向：
1. 邮箱验证 — 注册后发验证邮件，点击激活账号
2. 忘记密码 — 发重置密码邮件（nodemailer）
3. 第三方登录 — GitHub / Google OAuth（passport.js）
4. 多设备管理 — 每个设备一个 refresh token，可单独踢下线
5. 操作日志 — 记录登录 IP、时间、设备信息
6. 二步验证 — TOTP（Google Authenticator）
7. 账号封禁 — isActive 字段 + 封禁原因
```
